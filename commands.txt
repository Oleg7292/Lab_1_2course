M PUSH 10          // Добавить 10 в массив
M PUSH 20          // Добавить 20 в массив
M GET 0            // Получить элемент по индексу 0 (ожидается: 10)
M DEL 1            // Удалить элемент по индексу 1 (удаляет 20)

L PUSH 30          // Добавить 30 в односвязный список
L DEL 30           // Удалить 30 из односвязного списка

DL PUSH_FRONT 40   // Добавить 40 в начало двусвязного списка
DL PUSH_BACK 50    // Добавить 50 в конец двусвязного списка
DL DEL 40          // Удалить 40 из двусвязного списка

S PUSH 60          // Добавить 60 в стек
S PUSH 70          // Добавить 70 в стек
S POP              // Удалить верхний элемент стека

Q PUSH 80          // Добавить 80 в очередь
Q PUSH 90          // Добавить 90 в очередь
Q POP              // Удалить первый элемент из очереди

H SET mykey 100    // Установить значение 100 для ключа "mykey"
H GET mykey        // Получить значение по ключу "mykey" (ожидается: 100)
H DEL mykey        // Удалить элемент с ключом "mykey"

T INSERT 200       // Добавить 200 в полное бинарное дерево
T CHECK_COMPLETE    // Проверить, является ли дерево полным
